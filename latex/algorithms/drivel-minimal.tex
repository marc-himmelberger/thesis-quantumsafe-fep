	\NextLine

	\NextLine
	\ClientAction{if $\funPRF(ES, ect_e \conc \mathsf{auth} \conc P_S \conc M_S \conc \textlit{:mac\_s}) \neq \mathsf{MAC}_S$: $\texttt{break}$}

	\NextLine

	\NextLine

	\NextLine
